    (use-package lsp-mode
      :commands lsp
      :hook ((java-mode . lsp)
             (python-mode . lsp)

      (setq
       read-process-output-max (* 1024 1024)
       lsp-idle-delay 0.500
       lsp-pyls-plugins-pydocstyle-enabled t
